3.2.3 c05_02 x sexo x edad

# Crear data srvy y filtrar solo los que estan de acuerdo o muy de acuerdo
data_filtrada <- data_svy %>%
  filter(
    !is.na(sexo),
    !is.na(edad_tramos),
    !is.na(c05_02_acuerdo)
  ) %>%
  filter(c05_02_acuerdo == 1)

# Calcular total por sexo y tramo etario (solo quienes están de acuerdo)
acuerdo_por_grupo <- data_filtrada %>%
  group_by(sexo, edad_tramos) %>%
  summarise(total = survey_total(na.rm = TRUE), .groups = "drop")

# Calcular total por sexo (para normalizar a 100% dentro de cada sexo)
totales_por_sexo <- acuerdo_por_grupo %>%
  group_by(sexo) %>%
  summarise(total_sexo = sum(total), .groups = "drop")

# Unir y calcular proporciones relativas dentro de cada sexo
acuerdo_final <- acuerdo_por_grupo %>%
  left_join(totales_por_sexo, by = "sexo") %>%
  mutate(
    prop_sexo = total / total_sexo
  )

# Graficar: barras separadas por sexo (cada grupo suma 100%)
ggplot(acuerdo_final, aes(x = edad_tramos, y = prop_sexo, fill = sexo)) +
  geom_col(position = position_dodge(width = 0.8)) +
  geom_text(
    aes(label = scales::percent(prop_sexo, accuracy = 0.1)),
    position = position_dodge(width = 0.8),
    vjust = -0.3,
    size = 4,
    color = "black"
  ) +
  scale_fill_manual(values = c("Mujer" = "#f25f5c", "Hombre" = "#9e1b1e")) +
  scale_y_continuous(labels = scales::percent_format(accuracy = 1), expand = expansion(mult = c(0, 0.1))) +
  labs(
    title = "Acuerdo con: Una de las principales razones de la alta desigualdad en Chile \nes que los dueños de las empresas pagan poco a sus trabajadores",
    subtitle = "Porcentaje según sexo y tramo etario",
    caption = "Nota: Datos ponderados. Proporción que responde De acuerdo (4) o Muy de acuerdo (5). N = 3048.",
    x = "Tramo de edad",
    y = "Porcentaje",
    fill = "Sexo"
  ) +
  theme_minimal()

# Guardar gráfico
ggsave("output/c05_02_sexo_edad.png", width = 9, height = 6, dpi = 300)

    x = "Tramo de edad",
